-	script	Healer	-1,{

	.@Price = 0;	// ค่าใช้จ่ายในการรักษา
	.@Buffs = 1;	// บัฟผู้เล่นด้วย? (1: ใช่ / 0: ไม่)
	.@Delay = 0;	// ระยะเวลารักษาเป็นวินาที

	if (@HD > gettimetick(2))
		end;
	if (.@Price) {
		message strcharinfo(0), "ค่าใช้จ่ายในการรักษา " + callfunc("F_InsertComma",.@Price) + " Zeny";
		if (Zeny < .@Price)
			end;
		if (select("^0055FFรักษา^000000:^777777ยกเลิก^000000") == 2)
			end;
		Zeny -= .@Price;
	}
	specialeffect2 EF_HEAL2;
	percentheal 100,100;
	if (.@Buffs) {
	
		//บัพอาหาร 30 นาที เลเวล 10
		sc_start SC_STRFood, 1800000,10;
		sc_start SC_AGIFood, 1800000,10;
		sc_start SC_VITFood, 1800000,10;
		sc_start SC_INTFood, 1800000,10;
		sc_start SC_DEXFood, 1800000,10;
		sc_start SC_LUKFood, 1800000,10;
		
		//บัพสกิลต่างๆ
		sc_start SC_BLESSING,1800000,10;
		sc_start SC_INCREASEAGI,1800000,10;
		sc_start SC_GLORIA,1800000,10;
		sc_start SC_MAGNIFICAT,1800000,10;
		sc_start4 SC_POEMBRAGI,1800000,5,70,90,5;
		sc_start4 SC_SERVICE4U,1800000,5,40,40,1;
		
		//แก้สถานะต่างๆ
		sc_end SC_POISON;
		sc_end SC_FREEZE;
		sc_end SC_SLEEP;
		sc_end SC_STAN;
		sc_end SC_STUN;
		sc_end SC_STONE;
		sc_end SC_SILENCE;
		sc_end SC_BLIND;
		sc_end SC_Bleeding;
		sc_end SC_CONFUSION;
		sc_end SC_CURSE;
		sc_end SC_HALLUCINATION;
		sc_end SC_ORCISH;
		
	}
	if (.@Delay)
		@HD = gettimetick(2) + .@Delay;
	end;
}
